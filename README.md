# installation

  sudo apt install python3 pip git
  
  pip install pygame
  
  git clone https://github.com/al1oncik/BirdGame.git
  
  cd BirdGame
  
  python3 game.py
  
  

# Controlling

  key space, left mouse button -> jump
  
  r -> retry
